\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{arrows,shapes,shadows,positioning}

\tikzset{
  %Define standard arrow tip
    >=stealth',
  %define style for boxes
  punkt/.style={
    rectangle,
    rounded corners,
    draw=black,very thick,
    text width=7.5em,
    minimum height = 2em,
    text centered},
  punkp/.style={
    rectangle,
    dashed,
    rounded corners,
    draw=black,very thick,
    text width=7.5em,
    text centered, 
    anchor=north,
    minimum height = 2em},
  %define style for arrows
  myarrow/.style={
    ->, 
    >=open triangle 90, 
    shorten >=1pt, 
    thick}
}

\begin{tikzpicture}[node distance=1cm, auto,]
